#define TPCMaskViewDefaultAlpha 0.5
@interface TPCCaptureImageView()

/** 手势 */
@property (strong, nonatomic) UIPanGestureRecognizer *pan;

/** 起始位置 */
@property (assign, nonatomic) CGPoint startPoint;

/** 遮罩view */
@property (weak, nonatomic) UIView *maskView;

@end

@implementation TPCCaptureImageView

- (instancetype)initWithFrame:(CGRect)frame
{
    if (self = [super initWithFrame:frame]) {
        [self setUp];
    }

    return self;
}

- (instancetype)initWithCoder:(NSCoder *)coder
{
    self = [super initWithCoder:coder];
    if (self) {
        [self setUp];
    }
    return self;
}

- (void)setUp
{
    // 添加拖拽手势
    _pan = [[UIPanGestureRecognizer alloc] initWithTarget:self action:@selector(pan:)];
    [self addGestureRecognizer:_pan];

    // 默认开启用户交互
    self.userInteractionEnabled = YES;
}

#pragma mark 懒加载
- (UIView *)maskView
{
    if (_maskView == nil) {
        UIView *maskView = [[UIView alloc] init];
        maskView.backgroundColor = [UIColor blackColor];
        maskView.alpha = TPCMaskViewDefaultAlpha;
        [self addSubview:maskView];
        self.maskView = maskView;
    }

    return _maskView;
}

#pragma mark 手势回调函数
- (void)pan:(UIPanGestureRecognizer *)pan
{
    CGPoint currentPoint = [pan locationInView:self];

    if (pan.state == UIGestureRecognizerStateBegan) {
        self.startPoint = currentPoint;
    } else if (pan.state == UIGestureRecognizerStateChanged) {
        // 设置遮罩frame
        CGFloat maskW = currentPoint.x - self.startPoint.x;
        CGFloat maskH = currentPoint.y - self.startPoint.y;
        self.maskView.frame = CGRectMake(self.startPoint.x, self.startPoint.y, maskW, maskH);
    } else if (pan.state == UIGestureRecognizerStateEnded) {
        // 获取裁剪后的图片
        UIImage *image = [self captureImageWithRect:self.maskView.frame];

        if ([self.delegate respondsToSelector:@selector(captureImageView:capturedImage:)]) {
            // 输出裁剪后的图片，并获取要显示的图片
            self.image = [self.delegate captureImageView:self capturedImage:image];
        }
    }
}

#pragma mark 获取裁剪图片
- (UIImage *)captureImageWithRect:(CGRect)rect
{
    UIBezierPath *path = nil;

    // maskView使不可见
    self.maskView.alpha = 0;

    if (self.captureType == TPCCaptureTypeTrim) {
        // 格式1.裁剪
        // 开启图片上下文，尺寸为imageView
        UIGraphicsBeginImageContextWithOptions(self.bounds.size, NO, 0);

        // 设置裁剪区域
        path = [UIBezierPath bezierPathWithRect:rect];
        [path addClip];

        // 获取上下文
        CGContextRef ctx = UIGraphicsGetCurrentContext();
        // 进行截屏
        [self.layer renderInContext:ctx];

    } else if (self.captureType == TPCCaptureTypeStuff) {
        // 格式2.填充
        // 开启图片上下文，尺寸为maskView
        UIGraphicsBeginImageContextWithOptions(self.maskView.bounds.size, NO, 0);

        // 将imageView的图片重新绘制进图片上下文中
        // 将图片的原点偏移，使选中区域刚好在上下文规定的大小中
        [self.image drawInRect:CGRectMake(-self.maskView.frame.origin.x, -self.maskView.frame.origin.y, self.bounds.size.width, self.bounds.size.height)];
    }

    // 获取上下文中的图片
    UIImage *image = UIGraphicsGetImageFromCurrentImageContext();

    // 关闭图片上下文
    UIGraphicsEndImageContext();

    // maskView可见，并置为初始状态
    self.maskView.alpha = TPCMaskViewDefaultAlpha;
    self.maskView.frame = CGRectZero;

    return image;
}

- (void)setEnbleCapture:(BOOL)enbleCapture
{
    _enbleCapture = enbleCapture;

    // 设置手势使能喝用户交互使能
    self.pan.enabled = enbleCapture;
    self.userInteractionEnabled = enbleCapture;
}

layer上下文

  • -drawRect:方法
    • 只有在这个方法里面才能获取到跟view的layer相关联的图形上下文［仅限制为layer上下文］
      • 重新绘制注意点
        • 不能手动调用，因为手动点用时不能获取图形上下文，我们自己创建不了，只能系统帮我们创建，并且传递给我们
        • 要想重新绘制，需要调用setNeedsDisplay
          • 这个方法并不会马上调用drawRect，其实这个方法只是给当前控件添加刷新的标记，等下一次屏幕刷新的时候才会调用drawRect，所以多次调用setNeedsDisplay并不会造成性能损失
    • 调用时间
      • 当这个view要显示的时候才会调用drawRect绘制图形
        • 在viewWillAppear之后在viewDidAppear之前调用
      • 使用setNeedsDisplay后，在屏幕进行刷新时也会进行调用
    • rect参数
      • 这个参数传进来的是控件的bounds
    • 注意，只要调用了这个方法，就会把前面绘制的东西全部清除
    • 当渲染系统准备好，它会调用视图图层的-display方法，此时，图层会装配他的后备存储。然后建立一个CoreGraphics上下文(CGContextRef)，将后备存储对应内存中的数据恢复出来，绘图会进入对应的内存区域，并使用CGContextRef绘制。
    • 当使用UIKit的绘制方法如(UIRectFill()/[UIBezierPath fill])代替-drawRect方法，他们将会使用这个上下文。使用步骤是：UiKit将后备存储的CGContextRef推进他的上下文堆栈(貌似不管怎样，都会先push进管理的上下文堆栈)，也就是说，他会将那个上下文设置为当前的。因此UIGraphicsGetCurrent将会返回那个对应的上下文。如果想直接使用CoreGraphics方法，可以自己调用UIGraphicsGetCurrent得到相同的上下文，并且讲这个上下文传给CoreGraphics方法。
  • 图形上下文
    • 可以理解为画布，我的理解就是缓存，专门存放我要画到屏幕上的内容（因为CPU输送数据的速度总是快于屏幕显示的速度的，所以需要这么一个缓存）
    • 图形上下文必须跟view相关联，才能将内容绘制到view上面
    • 目前我们所用的上下文都是以UIGraphics开头进行获取
    • CGContextRef

    - (void)drawRect:(CGRect)rect
{
  // 获取随机数值数组
  NSArray *array = [self randomIntegerArrayWithin:100 elementNumber:4];

  CGPoint center = CGPointMake(rect.size.width * 0.5, rect.size.height * 0.5);
  CGFloat startAngle = 0;
  CGFloat endAngle = 0;

  for (NSNumber *number in array) {
      endAngle = startAngle + [number floatValue] / 100.0 * M_PI * 2;

      UIBezierPath *path = [UIBezierPath bezierPathWithArcCenter:center radius:rect.size.width * 0.5 startAngle:startAngle endAngle:endAngle clockwise:YES];
      [path addLineToPoint:center];

      // 获取随机颜色
      [[self randomColor] set];
      [path fill];

      // 下一个弧形的起始角度等于当前弧形的结束角度
      startAngle = endAngle;
  }
}

    - (void)drawRect:(CGRect)rect
{
  NSInteger count = 4;
  NSInteger max  = 100;

  // 获取随机数值数组
  NSArray *array = [self randomIntegerArrayWithin:max elementNumber:count];

  CGFloat w = rect.size.width / (array.count * 2 - 1);
  __block CGFloat x = 0;
  __block CGFloat y = 0;
  __block CGFloat h = 0;

  [array enumerateObjectsUsingBlock:^(id obj, NSUInteger idx, BOOL *stop) {
      x = w * idx * 2;
      h = [obj floatValue] / (CGFloat)max * rect.size.height;

      // 倒立着画
      y = rect.size.height - h;

      UIBezierPath *path = [UIBezierPath bezierPathWithRect:CGRectMake(x, y, w, h)];

      // 获取随机颜色
      [[self randomColor] set];
      [path fill];
  }];
}

CGContextScaleCTM(ctx, 0.5, 0.5);
CGContextRotateCTM(ctx, M_PI_4);

// 以下方法将路径添加进上下文，并且渲染
[path fill];
}
      • 而手动添加路径的原生的绘制需要在addPath之前写
      - (void)drawRect:(CGRect)rect
{
CGContextRef ctx = UIGraphicsGetCurrentContext();

CGMutablePathRef path = CGPathCreateMutable();

CGPathMoveToPoint(path, NULL, 0, 0);
CGPathAddLineToPoint(path, NULL, 200, 200);

// 需要在addPath之前写
CGContextRotateCTM(ctx, M_PI_4);

// 向图形上下文添加路径
CGContextAddPath(ctx, path);

CGContextStrokePath(ctx);
}
      • 因为需要在添加路径前进行矩阵操作，所以使用系统默认创建的路径就不行了，矩阵操作不能生效
        • 按理论，路径被加进上下文，是被拷贝内容过去的，也就是说，一旦添加路径后，即使我再修改原先添加的路径也不能生效的，需要再进行一次添加。但是以下代码却能对已经在上下文中的路径进行修改，可能CGContext开头的函数就是操作的上下文中的默认路径
      - (void)drawRect:(CGRect)rect
{
CGContextRef ctx = UIGraphicsGetCurrentContext();

// 应该在这一句就在上下文中创建了默认路径
CGContextMoveToPoint(ctx, 20, 20);

// 输出有值，说明默认路径在创建时就被加入了上下文中
NSLog(@"%@", NSStringFromCGPoint(CGContextGetPathCurrentPoint(ctx)));

CGContextAddLineToPoint(ctx, 200, 200);

// 系统默认的路径已经添加进默认上下文了，所以以下矩阵操作无效
CGContextRotateCTM(ctx, M_PI_4);

CGContextStrokePath(ctx);
}
  • UIBezierPath
    • 使用UIBezierPath为什么不传入一个上下文参数到方法中？因为当使用UIKit或者AppKit时，上下文是唯一的。UIKit维护着一个上下文堆栈，UIKit方法总是绘制到最顶层的上下文中。
    // 获取顶层上下文
UIGraphicsGetCurrentContext()
// 在UiKit的堆栈中推进上下文
UIGraphicsPushContext(<#CGContextRef context#>)
// 在Uikit的堆栈中取出上下文
UIGraphicsPopContext()
